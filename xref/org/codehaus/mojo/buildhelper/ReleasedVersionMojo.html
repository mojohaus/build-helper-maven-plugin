27  import java.util.List;
28  
29  import org.apache.maven.artifact.ArtifactUtils;
30  import org.apache.maven.artifact.factory.ArtifactFactory;
31  import org.apache.maven.artifact.metadata.ArtifactMetadataRetrievalException;
32  import org.apache.maven.artifact.metadata.ArtifactMetadataSource;
33  import org.apache.maven.artifact.repository.ArtifactRepository;
34  import org.apache.maven.artifact.versioning.ArtifactVersion;
35  import org.apache.maven.plugins.annotations.Component;
36  import org.apache.maven.plugins.annotations.LifecyclePhase;
37  import org.apache.maven.plugins.annotations.Mojo;
38  import org.apache.maven.plugins.annotations.Parameter;
39  
40  /**
41   * Resolve the latest released version of this project.
42   * This mojo sets the following properties:
43   *
44   * <pre>
45   *   [propertyPrefix].version
46   *   [propertyPrefix].majorVersion
47   *   [propertyPrefix].minorVersion
48   *   [propertyPrefix].incrementalVersion
49   * </pre>
50   *
51   * Where the propertyPrefix is the string set in the mojo parameter.
52   *
53   * @author Robert Scholte
54   * @since 1.6
55   */
56  @Mojo( name = "released-version", defaultPhase = LifecyclePhase.VALIDATE, threadSafe = true )
57  public class ReleasedVersionMojo
58      extends AbstractDefinePropertyMojo
59  {
60  
61      /**
62       * The artifact metadata source to use.
63       */
64      @Component
65      private ArtifactMetadataSource artifactMetadataSource;
66  
67      @Component
68      private ArtifactFactory artifactFactory;
69  
70      @Parameter( defaultValue = "${localRepository}", readonly = true )
71      private ArtifactRepository localRepository;
72  
73      @Parameter( defaultValue = "${project.remoteArtifactRepositories}", readonly = true )
74      private List<ArtifactRepository> remoteArtifactRepositories;
75  
76      /**
77       * Prefix string to use for the set of version properties.
78       */
79      @Parameter( defaultValue = "releasedVersion" )
80      private String propertyPrefix;
81  
82      private void defineVersionProperty( String name, String value )
83      {
84          defineProperty( propertyPrefix + '.' + name, value );
85      }
86  
87      private void defineVersionProperty( String name, int value )
88      {
89          defineVersionProperty( name, Integer.toString( value ) );
90      }
91  
92      @SuppressWarnings( "unchecked" )
93      public void execute()
94      {
95          org.apache.maven.artifact.Artifact artifact =
96              artifactFactory.createArtifact( getProject().getGroupId(), getProject().getArtifactId(), "", "", "" );
97          try
98          {
99              ArtifactVersion releasedVersion = null;
100             List<ArtifactVersion> versions =
101                 artifactMetadataSource.retrieveAvailableVersions( artifact, localRepository, remoteArtifactRepositories );
102             for ( ArtifactVersion version : versions )
103             {
104                 if ( !ArtifactUtils.isSnapshot( version.toString() )
105                     && ( releasedVersion == null || version.compareTo( releasedVersion ) > 0 ) )
106                 {
107                     releasedVersion = version;
108                 }
109             }
110 
111             if ( releasedVersion != null )
112             {
113                 // Use ArtifactVersion.toString(), the major, minor and incrementalVersion return all an int.
114                 String releasedVersionValue = releasedVersion.toString();
115 
116                 // This would not always reflect the expected version.
117                 int dashIndex = releasedVersionValue.indexOf( '-' );
118                 if ( dashIndex >= 0 )
119                 {
120                     releasedVersionValue = releasedVersionValue.substring( 0, dashIndex );
121                 }
122 
123                 defineVersionProperty( "version", releasedVersionValue );
124                 defineVersionProperty( "majorVersion", releasedVersion.getMajorVersion() );
125                 defineVersionProperty( "minorVersion", releasedVersion.getMinorVersion() );
126                 defineVersionProperty( "incrementalVersion", releasedVersion.getIncrementalVersion() );
127             }
128 
129         }
130         catch ( ArtifactMetadataRetrievalException e )
131         {
132             if ( getLog().isWarnEnabled() )
133             {
134                 getLog().warn( "Failed to retrieve artifacts metadata, cannot resolve the released version" );
135             }
136         }
137     }
138 }
