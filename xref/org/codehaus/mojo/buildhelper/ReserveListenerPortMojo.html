52  /**
53   * Reserve a list of random and not in use network ports and place them in a configurable project properties.
54   *
55   * @author <a href="dantran@gmail.com">Dan T. Tran</a>
56   * @version $Id: ReserveListnerPortMojo.java 6754 2008-04-13 15:14:04Z dantran $
57   * @since 1.2
58   */
59  @Mojo( name = "reserve-network-port", defaultPhase = LifecyclePhase.PROCESS_TEST_CLASSES, threadSafe = true )
60  public class ReserveListenerPortMojo
61      extends AbstractMojo
62  {
63      private static final String BUILD_HELPER_RESERVED_PORTS = "BUILD_HELPER_MIN_PORT";
64  
65      private static final Integer FIRST_NON_ROOT_PORT_NUMBER = 1024;
66  
67      private static final Integer MAX_PORT_NUMBER = 65536;
68  
69      private static final Object lock = new Object();
70  
71      /**
72       * A List to property names to be placed in Maven project. At least one of {@code #urls} or {@code #portNames} has to be specified.
73       *
74       * @since 1.2
75       */
76      @Parameter
77      private String[] portNames = new String[0];
78  
79      /**
80       * A List of urls to resource where list of name could be found. Can be in
81       * form of classpath:com/myproject/names.txt . At least one of {@code #urls} or {@code #portNames} has to be specified.
82       * 
83       * @since 1.11
84       */
85      @Parameter
86      private String[] urls = new String[0];
87  
88      /**
89       * Output file to write the generated properties to. if not given, they are written to Maven project
90       *
91       * @since 1.2
92       */
93      @Parameter
94      private File outputFile;
95  
96      /**
97       * Specify this if you want the port be chosen with a number higher than that one.
98       * <p>
99       * If {@link #maxPortNumber} is specified, defaults to {@value #FIRST_NON_ROOT_PORT_NUMBER}.
100      * </p>
101      *
102      * @since 1.8
103      */
104     @Parameter
105     private Integer minPortNumber;
106 
107     /**
108      * Specify this if you want the port be chosen with a number lower than that one.
109      *
110      * @since 1.8
111      */
112     @Parameter
113     private Integer maxPortNumber;
114 
115     /**
116      * Specify true or false if you want the port selection randomized.
117      * <p>
118      * </p>
119      *
120      * @since 1.10
121      */
122     @Parameter
123     private boolean randomPort;
124 
125     /**
126      * @since 1.2
127      */
128     @Parameter( readonly = true, defaultValue = "${project}" )
129     private MavenProject project;

131     @Override
132     public void execute()
133         throws MojoExecutionException
134     {
135         Properties properties = project.getProperties();
136 
137         if ( outputFile != null )
138         {
139             properties = new Properties();
140         }
141 
142         loadUrls();
143 
144         // Reserve the entire block of ports to guarantee we don't get the same port twice
145         final List<ServerSocket> sockets = new ArrayList<ServerSocket>();
146         try
147         {
148             for ( String portName : portNames )
149             {
150                 try
151                 {
152                     final ServerSocket socket = getServerSocket();
153                     sockets.add( socket );
154 
155                     final String unusedPort = Integer.toString( socket.getLocalPort() );
156                     properties.put( portName, unusedPort );
157                     getReservedPorts().add( socket.getLocalPort() );
158                     this.getLog().info( "Reserved port " + unusedPort + " for " + portName );
159                 }
160                 catch ( IOException e )
161                 {
162                     throw new MojoExecutionException( "Error getting an available port from system", e );
163                 }
164             }
165 
166             // Write the file -- still hold onto the ports
167             if ( outputFile != null )
168             {
169                 OutputStream os = null;
170                 try
171                 {
172                     os = new FileOutputStream( outputFile );
173                     properties.store( os, null );
174                 }
175                 catch ( Exception e )
176                 {
177                     throw new MojoExecutionException( e.getMessage() );
178                 }
179                 finally
180                 {
181                     IOUtil.close( os );
182                 }
183             }
184         }
185         finally
186         {
187             // Now free all the ports
188             for ( ServerSocket socket : sockets )
189             {
190                 final int localPort = socket.getLocalPort();
191                 try
192                 {
193                     socket.close();
194                 }
195                 catch ( IOException e )
196                 {
197                     this.getLog().error( "Cannot free reserved port " + localPort );
198                 }
199             }
200         }
201     }
202 
203     private ServerSocket getServerSocket()
204         throws IOException, MojoExecutionException
205     {
206         if ( minPortNumber == null && maxPortNumber != null )
207         {
208             getLog().debug( "minPortNumber unspecified: using default value " + FIRST_NON_ROOT_PORT_NUMBER );
209             minPortNumber = FIRST_NON_ROOT_PORT_NUMBER;
210         }
211         if ( minPortNumber != null && maxPortNumber == null )
212         {
213             getLog().debug( "maxPortNumber unspecified: using default value " + MAX_PORT_NUMBER );
214             maxPortNumber = MAX_PORT_NUMBER;
215         }
216         if ( minPortNumber == null && maxPortNumber == null )
217         {
218             return new ServerSocket( 0 );
219         }
220         if ( randomPort )
221         {
222             synchronized ( lock )
223             {
224                 List<Integer> availablePorts = randomPortList();
225                 for ( Iterator<Integer> iterator = availablePorts.iterator(); iterator.hasNext(); )
226                 {
227                     int port = iterator.next();
228                     ServerSocket serverSocket = reservePort( port );
229                     iterator.remove();
230                     if ( serverSocket != null )
231                     {
232                         return serverSocket;
233                     }
234                 }
235                 throw new MojoExecutionException( "Unable to find an available port between " + minPortNumber + " and "
236                     + maxPortNumber );
237             }
238         }
239         else
240         {
241             // Might be synchronizing a bit too largely, but at least that defensive approach should prevent
242             // threading issues (essentially possible while put/getting the plugin ctx to get the reserved ports).
243             synchronized ( lock )
244             {
245                 int min = getNextPortNumber();
246 
247                 for ( int port = min;; ++port )
248                 {
249 
250                     ServerSocket serverSocket = reservePort( port );
251                     if ( serverSocket != null ) {
252                         return serverSocket;
253                     }
254                 }
255             }
256         }
257     }
258 
259     private List<Integer> randomPortList()
260     {
261 
262         int difference = maxPortNumber - minPortNumber + 1;
263         List<Integer> portList = new ArrayList<Integer>( difference );
264         List<Integer> reservedPorts = getReservedPorts();
265         for ( int i = 0; i < difference; i++ )
266         {
267             int port = minPortNumber + i;
268             if ( !reservedPorts.contains( port ) )
269             {
270                 portList.add( minPortNumber + i );
271             }
272         }
273         Collections.shuffle( portList );
274         return portList;
275     }

277     public ServerSocket reservePort( int port )
278         throws MojoExecutionException
279     {
280 
281         if ( port > maxPortNumber )
282         {
283             throw new MojoExecutionException( "Unable to find an available port between " + minPortNumber + " and "
284                 + maxPortNumber );
285         }
286         try
287         {
288             ServerSocket serverSocket = new ServerSocket( port );
289             getLog().info( "Port assigned" + port );
290             return serverSocket;
291         }
292         catch ( IOException ioe )
293         {
294             getLog().info( "Tried binding to port " + port + " without success. Trying next port.", ioe );
295         }
296         return null;
297     }
298 
299     private int getNextPortNumber()
300     {
301         assert minPortNumber != null;
302 
303         List<Integer> reservedPorts = getReservedPorts();
304         int nextPort = -1;
305         if ( reservedPorts.isEmpty() )
306         {
307             nextPort = minPortNumber;
308         }
309         else
310         {
311             nextPort = findAvailablePortNumber( minPortNumber, reservedPorts );
312         }
313         reservedPorts.add( nextPort );
314         getLog().debug( "Next port: " + nextPort );
315         return nextPort;
316     }
317 
318     @SuppressWarnings( "unchecked" )
319     private List<Integer> getReservedPorts()
320     {
321 
322         List<Integer> reservedPorts = (List<Integer>) getPluginContext().get( BUILD_HELPER_RESERVED_PORTS );
323         if ( reservedPorts == null )
324         {
325             reservedPorts = new ArrayList<Integer>();
326             getPluginContext().put( BUILD_HELPER_RESERVED_PORTS, reservedPorts );
327         }
328         return reservedPorts;
329     }

331     /**
332      * Returns the first number available, starting at portNumberStartingPoint that's not already in the reservedPorts
333      * list.
334      *
335      * @param portNumberStartingPoint first port number to start from.
336      * @param reservedPorts the ports already reserved.
337      * @return first number available not in the given list, starting at the given parameter.
338      */
339     private int findAvailablePortNumber( Integer portNumberStartingPoint, List<Integer> reservedPorts )
340     {
341         assert portNumberStartingPoint != null;
342         int candidate = portNumberStartingPoint;
343         while ( reservedPorts.contains( candidate ) )
344         {
345             candidate++;
346         }
347         return candidate;
348     }

350     private void loadUrls() throws MojoExecutionException {
351         for (String url : urls) {
352             load(new UrlResource(url));
353         }
354     }
355 
356     private void load(UrlResource resource) throws MojoExecutionException {
357         if (resource.canBeOpened()) {
358             loadPortNamesFromResource(resource);
359         } else {
360             throw new MojoExecutionException("Port names could not be loaded from \"" + resource + "\"");
361         }
362     }
363 
364     private void loadPortNamesFromResource(UrlResource resource) throws MojoExecutionException {
365         try {
366             if (getLog().isDebugEnabled()) {
367                 getLog().debug("Loading port names from " + resource);
368             }
369             final InputStream stream = resource.getInputStream();
370 
371             try {
372                 BufferedReader reader = new BufferedReader(new InputStreamReader(stream));
373                 List<String> names = new ArrayList<String>();
374                 String line;
375                 while ((line = reader.readLine()) != null) {
376                     line = line.trim();
377                     if (!line.isEmpty() && !line.replace(" ", "").startsWith("#")) {
378                         names.add(line);
379                     }
380                 }
381                 if (getLog().isDebugEnabled()) {
382                     getLog().debug("Loaded port names " + names);
383                 }
384                 String[] tPortNames = names.toArray(new String[portNames.length + names.size()]);
385                 if (portNames.length > 0) {
386                     System.arraycopy(portNames, 0, tPortNames, names.size(), portNames.length);
387                 }
388                 portNames = tPortNames;
389             } finally {
390                 stream.close();
391             }
392         } catch (IOException e) {
393             throw new MojoExecutionException("Error reading port names from \"" + resource + "\"", e);
394         }
395     }

397     private class UrlResource {
398         private static final String CLASSPATH_PREFIX = "classpath:";
399 
400         private static final String SLASH_PREFIX = "/";
401 
402         private final URL url;
403 
404         private boolean isMissingClasspathResouce = false;
405 
406         private String classpathUrl;
407 
408         private InputStream stream;
409 
410         public UrlResource(String url)
411                 throws MojoExecutionException {
412             if (url.startsWith(CLASSPATH_PREFIX)) {
413                 String resource = url.substring(CLASSPATH_PREFIX.length(), url.length());
414                 if (resource.startsWith(SLASH_PREFIX)) {
415                     resource = resource.substring(1, resource.length());
416                 }
417                 this.url = getClass().getClassLoader().getResource(resource);
418                 if (this.url == null) {
419                     if (getLog().isDebugEnabled()) {
420                         getLog().debug("Can not load classpath resouce \"" + url + "\"");
421                     }
422                     isMissingClasspathResouce = true;
423                     classpathUrl = url;
424                 }
425             } else {
426                 try {
427                     this.url = new URL(url);
428                 } catch (MalformedURLException e) {
429                     throw new MojoExecutionException("Badly formed URL " + url + " - " + e.getMessage());
430                 }
431             }
432         }
433 
434         public InputStream getInputStream() throws IOException {
435             if (stream == null) {
436                 stream = openStream();
437             }
438             return stream;
439         }
440 
441         public boolean canBeOpened() {
442             if (isMissingClasspathResouce) {
443                 return false;
444             }
445             try {
446                 openStream().close();
447             } catch (IOException e) {
448                 return false;
449             }
450             return true;
451         }
452 
453         private InputStream openStream() throws IOException {
454             return new BufferedInputStream(url.openStream());
455         }
456 
457         @Override
458         public String toString() {
459             if (!isMissingClasspathResouce) {
460                 return "URL " + url.toString();
461             }
462             return classpathUrl;
463         }
464     }
