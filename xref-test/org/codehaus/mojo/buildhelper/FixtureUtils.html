22  import java.io.File;
23  import java.io.FileReader;
24  import java.io.IOException;
25  import java.util.Properties;
26  
27  /**
28   * A utility class for managing test resources.
29   *
30   * @author Adrian Price <a href="mailto:demonfiddler@virginmedia.com">demonfiddler@virginmedia.com</a>
31   * @since 1.12
32   */
33  public final class FixtureUtils
34  {
35      /**
36       * Creates test resources defined by the specification file <code>fixture.properties</code>. The method creates all
37       * the files named by the properties, optionally setting the modification timestamp by adding the number of seconds
38       * specified by the property value (which must be an integer) to the current system time.
39       * 
40       * @param basePath The path to the base directory against which to resolve relative paths (including
41       *            <code>fixture.properties</code>).
42       * @throws IOException If unable to create a file or a missing parent directory.
43       * @throws NumberFormatException If a non-empty property value cannot be parsed as an integer.
44       */
45      public static void createResources( String basePath )
46          throws IOException
47      {
48          File baseDir = new File( basePath );
49          createResources( baseDir, "fixture.properties", "UTF-8", System.currentTimeMillis() );
50      }
51  
52      /**
53       * Creates test resources defined by a specification file. The method creates all the files named by the properties,
54       * optionally setting the modification timestamp by adding the number of seconds specified by the property value
55       * (which must be an integer) to a base timestamp.
56       * 
57       * @param baseDir The base directory against which to resolve relative paths.
58       * @param propertiesPath The path (relative or absolute) to a <code>Properties</code> file whose property names are
59       *            file paths (relative or absolute) and whose optional property values represent relative modification
60       *            timestamps in seconds.
61       * @param baseTimestamp The base timestamp to use, in milliseconds since midnight on the 1st January 1970 GMT.
62       * @param encoding Encoding.
63       * @throws IOException If unable to create a file or a missing parent directory.
64       * @throws NumberFormatException If a non-empty property value cannot be parsed as an integer.
65       */
66      public static void createResources( File baseDir, String propertiesPath, String encoding, long baseTimestamp )
67          throws IOException
68      {
69          createResources( baseDir, new File( baseDir, propertiesPath ), encoding, baseTimestamp );
70      }
71  
72      /**
73       * Creates test resources defined by a specification file. The method creates all the files named by the properties,
74       * optionally setting the modification timestamp by adding the number of seconds specified by the property value
75       * (which must be an integer) to a base timestamp.
76       * 
77       * @param baseDir The base directory against which to resolve relative paths.
78       * @param propertiesFile A <code>Properties</code> file whose property names are file paths (relative or absolute)
79       *            and whose optional property values represent relative modification timestamps in seconds.
80       * @param encoding Encoding.
81       * @param baseTimestamp The base timestamp to use, in milliseconds since midnight on the 1st January 1970 GMT.
82       * @throws IOException If unable to create a file or a missing parent directory.
83       * @throws NumberFormatException If a non-empty property value cannot be parsed as an integer.
84       */
85      public static void createResources( File baseDir, File propertiesFile, String encoding, long baseTimestamp )
86          throws IOException
87      {
88          Properties properties = new Properties();
89          properties.load( new FileReader( propertiesFile ) );
90          createResources( baseDir, properties, baseTimestamp );
91      }
92  
93      /**
94       * Creates test resources defined by a specification file. The method creates all the files named by the properties,
95       * optionally setting the modification timestamp by adding the number of seconds specified by the property value
96       * (which must be an integer) to a base timestamp.
97       * 
98       * @param baseDir The base directory against which to resolve relative paths.
99       * @param properties A <code>Properties</code> object whose property names are file paths (relative or absolute) and
100      *            whose optional property values represent relative modification timestamps in seconds.
101      * @param baseTimestamp The base timestamp to use, in milliseconds since midnight on the 1st January 1970 GMT.
102      * @throws IOException If unable to create a file or a missing parent directory.
103      * @throws NumberFormatException If a non-empty property value cannot be parsed as an integer.
104      */
105     public static void createResources( File baseDir, Properties properties, long baseTimestamp )
106         throws IOException
107     {
108         for ( String propertyName : properties.stringPropertyNames() )
109         {
110             File file = new File( baseDir, propertyName );
111             File parent = file.getParentFile();
112             if ( parent != null )
113                 parent.mkdirs();
114             file.createNewFile();
115 
116             // If required, set the last modification timestamp.
117             String propertyValue = properties.getProperty( propertyName );
118             if ( !propertyValue.isEmpty() )
119             {
120                 int offset = Integer.parseInt( propertyValue );
121                 long ts = baseTimestamp + ( offset * 1000 );
122                 file.setLastModified( ts );
123                 assert file.lastModified() == ts : "failed to set last modified timestamp for "
124                     + file.getCanonicalPath();
125             }
126         }
127     }
128 
129     private FixtureUtils()
130     {
131     }
